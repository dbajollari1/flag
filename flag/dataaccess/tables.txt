1. Event
-------------
CREATE TABLE IF NOT EXISTS "event" (
"id" INTEGER PRIMARY KEY AUTOINCREMENT,
"title" VARCHAR,
"description" TEXT,
"location" VARCHAR,
"eventDate" DATE,
"startTime" VARCHAR,
"endTime" VARCHAR,
"status" VARCHAR,
"createdBy" VARCHAR, 
"createdOn" DATETIME
);
